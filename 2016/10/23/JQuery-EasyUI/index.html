JQuery EasyUI

1.下载easyUI开发包。

2.将easyUI资源文件引入到页面中。

easyui.css、icon.css、jquery-1.8.3.js、jquery.easyui.min.js

3.简单入门

3.1 layout页面布局。

<body class="easyui-layout">
    <!-- 使用div元素描述每个区域 -->
    <div style="height: 100px" data-options="region:'north'">北部区域</div>
    <div style="width: 200px" data-options="region:'west'">西部区域</div>
    <div data-options="region:'center'">中心区域</div>
    <div style="width: 100px" data-options="region:'east'">东部区域</div>
    <div style="height: 50px" data-options="region:'south'">南部区域</div>
</body>

3.2 accordion折叠面板。

<!-- accordion折叠面板 fit:true----自适应(填充父容器) -->
    <div class="easyui-accordion" data-options="fit:true">
        <!-- 使用子div表示每个面板 -->
        <div data-options="iconCls:'icon-cut'" title="面板一">内容1</div>
        <div title="面板二">内容2</div>
        <div title="面板三">内容3</div>
    </div>

3.3 tabs选项卡面板。

<!-- tabs选项卡面板 -->
<div class="easyui-tabs" data-options="fit:true">
    <!-- 使用子div表示每个面板 -->
    <div data-options="iconCls:'icon-cut'" title="面板一">内容1</div>
    <div data-options="closable:true" title="面板二">内容2</div>
    <div title="面板三">内容3</div>
</div>

4.JQuery EasyUI 动态添加选项卡。

方法：add 参数：options 
Example:
$('#mytabs').tabs({
        title:'标题',
        selected:'false'
        //...
    });
方法：select 选择一个选项卡面板,‘which’参数可以是标题或索引的选项卡面板
$("#mytabs").tabs("select","系统管理");
效果：选中当前选中行。
方法：exists 
var e = $("#mytabs").tabs("exists","系统管理");
效果：e为true即选项卡存在。false即不存在。

5.JQuery ztree插件。

5.1页面导入ztree相关文件。
jquery-1.4.4.min.js、zTreeStyle.css、jquery.ztree.core-3.5.js

5.2使用简单json数据构造ztree。

<ul id="tree" class="ztree"></ul>
<script type="text/javascript">
    $(function(){
        //页面加载完成后，执行下列代码，动态创建ztree
        var setting2 = {
                        data: {
                            simpleData: {
                                enable: true//表明要用简单json数据构造ztree节点
                                //简单json数据和非简单json数据区别在于构造节点数据时子节点构造方式
                            }
                        }
        };
        //构造节点数据
        var zNodes2 = [
                          {"id":"1","pId":"2","name":"节点一"},//每个json对象表示一个节点数据
                          {"id":"2","pId":"3","name":"节点二"},
                          //pId节点数据中保存其父节点唯一标识的属性名称
                          {"id":"3","pId":"0","name":"节点三"} //id：节点数据中保存唯一标识的属性名称

                        //使用非简单json数据格式构造ztree
                        //{"name":"节点一","children":[{"name":"节点一_1"}, {"name":"节点一_2"} ]}
                      ];
        //调用API初始化ztree
        $.fn.zTree.init($("#ztree"), setting, zNodes);

    });
</script>

5.3发送ajax请求获取json构造ztree

//发送ajax请求，获取json数据
//jQuery提供 的ajax方法：ajax、post、get、load、getJSON、getScript
    var url = "${pageContext.request.contextPath}/json/menu.json";
        $.post(url,{},function(data){
            //调用API初始化ztree

        },'json');
    });
